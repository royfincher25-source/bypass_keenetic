# Проверка ссылок на старый репозиторий

echo ""

FILES_WITH_OLD_LINKS=$(grep -rl "g00se72/bypass_keenetic" *.md 2>/dev/null)

if [ -n "$FILES_WITH_OLD_LINKS" ]; then
    echo "Найдены файлы со старыми ссылками:"
    echo "$FILES_WITH_OLD_LINKS"
    echo ""
    echo "Кол-во файлов: $(echo "$FILES_WITH_OLD_LINKS" | wc -l)"
    echo "Кол-во ссылок: $(grep -r "g00se72/bypass_keenetic" *.md | wc -l)"

    echo "✅ Все ссылки актуальны!"
fi
